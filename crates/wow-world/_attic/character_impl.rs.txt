    session.send_packet(&ResumeCommsResponse {});

/// Handle enum characters request.
pub async fn handle_enum_characters(session: &mut WorldSession) {
    use wow_packet::packets::character::EnumCharactersResult;
    
    debug!("Handling EnumCharacters for account {}", session.account_id);
    
    // Get characters from database
    let chars = match session.char_db() {
        Some(db) => {
            match db.query_characters(session.account_id).await {
                Ok(rows) => rows,
                Err(e) => {
                    warn!("Failed to query characters: {}", e);
                    Vec::new()
                }
            }
        }
        None => {
            warn!("No character database available");
            Vec::new()
        }
    };
    
    session.send_packet(&EnumCharactersResult { characters: chars });
}

/// Handle create character request.
pub async fn handle_create_character(session: &mut WorldSession, create: CreateCharacter) {
    use wow_packet::packets::character::{CreateCharacterResponse, CharacterInfo};
    
    debug!("Handling CreateCharacter for account {}: {}", session.account_id, create.name);
    
    // Validate inputs
    if create.name.is_empty() || create.name.len() > 12 {
        session.send_packet(&CreateCharacterResponse { result: 0x34, guid: ObjectGuid::EMPTY }); // CHAR_NAME_INVALID
        return;
    }
    
    // Create character in database
    let result = match session.char_db() {
        Some(db) => {
            match db.create_character(session.account_id, &create).await {
                Ok(guid) => {
                    info!("Created character {} with GUID {:?}", create.name, guid);
                    session.send_packet(&CreateCharacterResponse { result: 0x00, guid });
                    return;
                }
                Err(e) => {
                    warn!("Failed to create character: {}", e);
                    0x31 // CHAR_CREATE_ERROR
                }
            }
        }
        None => {
            warn!("No character database available");
            0x31
        }
    };
    
    session.send_packet(&CreateCharacterResponse { result: result, guid: ObjectGuid::EMPTY });
}

/// Handle character delete request.
pub async fn handle_char_delete(session: &mut WorldSession, delete: CharDelete) {
    use wow_packet::packets::character::DeleteCharResponse;
    
    debug!("Handling CharDelete for account {}: {:?}", session.account_id, delete.guid);
    
    // Verify ownership and delete
    let result = match session.char_db() {
        Some(db) => {
            match db.delete_character(session.account_id, delete.guid).await {
                Ok(_) => {
                    info!("Deleted character {:?}", delete.guid);
                    0x00 // CHAR_DELETE_SUCCESS
                }
                Err(e) => {
                    warn!("Failed to delete character: {}", e);
                    0x3C // CHAR_DELETE_FAILED
                }
            }
        }
        None => 0x3C,
    };
    
    session.send_packet(&DeleteCharResponse { result });
}

/// Handle player login request.
pub async fn handle_player_login(session: &mut WorldSession, login: PlayerLogin) {
    use wow_packet::packets::misc::NewWorld;
    
    debug!("Handling PlayerLogin for account {}: {:?}", session.account_id, login.guid);
    
    // Validate character ownership
    if !session.is_legit_character(&login.guid) {
        warn!("Player login rejected: {:?} not in legit characters", login.guid);
        session.state = SessionState::Disconnecting;
        return;
    }
    
    // Load character data
    let char_data = match session.char_db() {
        Some(db) => {
            match db.load_character_data(login.guid).await {
                Ok(data) => data,
                Err(e) => {
                    warn!("Failed to load character data: {}", e);
                    session.state = SessionState::Disconnecting;
                    return;
                }
            }
        }
        None => {
            warn!("No character database available");
            return;
        }
    };
    
    // Store player info
    session.player_guid = Some(login.guid);
    session.player_name = Some(char_data.name.clone());
    session.player_race = char_data.race;
    session.player_class = char_data.class;
    session.player_level = char_data.level;
    session.player_gender = char_data.gender;
    session.current_map_id = char_data.map_id;
    session.player_position = Some(Position {
        x: char_data.x,
        y: char_data.y,
        z: char_data.z,
        orientation: char_data.o,
    });
    session.player_gold = char_data.money;
    
    // Send NewWorld to client (map transfer)
    session.send_packet(&NewWorld {
        map_id: char_data.map_id as u32,
        x: char_data.x,
        y: char_data.y,
        z: char_data.z,
        orientation: char_data.o,
        reason: 0,
    });
    
    // Set state to transfer (waiting for client to acknowledge)
    session.state = SessionState::Transfer;
    
    debug!("Player login initiated for {} ({})", char_data.name, login.guid);
}

/// Handle ConnectTo failed.
pub async fn handle_connect_to_failed(session: &mut WorldSession, failed: ConnectToFailed) {
    warn!("ConnectTo failed for account {}: serial={:?}, con={}", 
        session.account_id, failed.serial, failed.con);
    
    // Reset connection state
    session.set_player_loading(None);
    session.set_connect_to_key(None);
}

/// Handle server time offset request.
pub async fn handle_server_time_offset_request(session: &mut WorldSession) {
    use wow_packet::packets::misc::ServerTimeOffsetResponse;
    use std::time::{SystemTime, UNIX_EPOCH};
    
    let now = SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .unwrap_or_default()
        .as_secs() as u32;
    
    session.send_packet(&ServerTimeOffsetResponse {
        time: now,
        offset: 0, // Server time offset (0 for same timezone)
    });
}

/// Handle played time request.
pub async fn handle_request_played_time(session: &mut WorldSession, trigger: bool) {
    use wow_packet::packets::misc::PlayedTimeResponse;
    use std::time::Instant;
    
    // Calculate current session time
    let session_time = if let Some(login_time) = session.login_time {
        Instant::now().duration_since(login_time).as_secs() as u32
    } else {
        0
    };
    
    let total = session.total_played_time + session_time;
    let level = session.level_played_time + session_time;
    
    session.send_packet(&PlayedTimeResponse {
        total_played_time: total,
        level_played_time: level,
        trigger_script_event: trigger,
    });
}

/// Handle logout request.
pub async fn handle_logout_request(session: &mut WorldSession, req: wow_packet::packets::misc::LogoutRequest) {
    use wow_packet::packets::misc::LogoutResponse;
    use std::time::{Duration, Instant};
    
    debug!("Logout requested for account {} (reason={})", session.account_id, req.reason);
    
    // Check if in combat
    if session.in_combat {
        session.send_packet(&LogoutResponse { result: 0x01, reason: 0 }); // LOGOUT_DENIED_COMBAT
        return;
    }
    
    // Start logout timer (20 seconds, or instant if requested)
    let logout_delay = if req.instant {
        Duration::from_secs(0)
    } else {
        Duration::from_secs(20)
    };
    
    session.logout_time = Some(Instant::now() + logout_delay);
    
    session.send_packet(&LogoutResponse { result: 0x00, reason: req.reason });
    
    debug!("Logout scheduled for account {} in {:?}", session.account_id, logout_delay);
}

/// Handle logout cancel.
pub async fn handle_logout_cancel(session: &mut WorldSession) {
    use wow_packet::packets::misc::LogoutCancelAck;
    
    session.logout_time = None;
    session.send_packet(&LogoutCancelAck {});
    
    debug!("Logout cancelled for account {}", session.account_id);
}
