    async fn handle_logout_request(&mut self, req: LogoutRequest) {
        use std::time::{Duration, Instant};
        
        debug!("Logout requested for account {} (reason={})", self.account_id, req.reason);
        
        // Check if in combat
        if self.in_combat {
            use wow_packet::packets::misc::LogoutResponse;
            self.send_packet(&LogoutResponse { result: 0x01, reason: 0 });
            return;
        }
        
        // Start logout timer
        let logout_delay = if req.instant {
            Duration::from_secs(0)
        } else {
            Duration::from_secs(20)
        };
        
        self.logout_time = Some(Instant::now() + logout_delay);
        
        use wow_packet::packets::misc::LogoutResponse;
        self.send_packet(&LogoutResponse { result: 0x00, reason: req.reason });
    }
