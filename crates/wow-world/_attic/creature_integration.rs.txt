    /// Check if a creature exists and is visible to this session.
    /// Uses the global MapManager for lookup.
    pub(crate) fn is_creature_visible(&self, guid: &ObjectGuid) -> bool {
        let Some(pos) = self.player_position else { return false; };
        let Some(map_manager) = &self.map_manager else { return false; };
        
        let map_guard = map_manager.read().unwrap();
        let creatures = map_guard.get_visible_creatures(
            self.current_map_id, 
            0, 
            pos.x, pos.y, pos.z
        );
        
        creatures.iter().any(|c| c.guid == *guid)
    }

    /// Spawn a creature into the global MapManager.
    pub(crate) fn spawn_creature_global(&self, creature: WorldCreature) -> bool {
        let Some(map_manager) = &self.map_manager else { return false; };
        
        let mut map_guard = map_manager.write().unwrap();
        let grid_x = world_to_grid_x(creature.position.x);
        let grid_y = world_to_grid_y(creature.position.y);
        
        let result = map_guard.add_creature(
            self.current_map_id,
            0,
            grid_x,
            grid_y,
            creature.clone()
        );
        
        if result {
            debug!("Spawned creature {:?} into global map {} at grid ({}, {})",
                creature.guid, self.current_map_id, grid_x, grid_y);
        }
        
        result
    }

    /// Update creature health in the global MapManager.
    pub(crate) fn update_creature_health_global(&self, guid: &ObjectGuid, new_hp: u32) -> bool {
        let Some(pos) = self.player_position else { return false; };
        let Some(map_manager) = &self.map_manager else { return false; };
        
        let mut map_guard = map_manager.write().unwrap();
        let grid_x = world_to_grid_x(pos.x);
        let grid_y = world_to_grid_y(pos.y);
        
        map_guard.with_creature_mut(
            self.current_map_id,
            0,
            grid_x,
            grid_y,
            *guid,
            |creature| {
                creature.current_hp = new_hp;
            }
        ).is_some()
    }

    /// Damage a creature in the global MapManager.
    /// Returns the new HP after damage.
    pub(crate) fn damage_creature_global(&self, guid: &ObjectGuid, damage: u32) -> Option<u32> {
        let Some(pos) = self.player_position else { return None; };
        let Some(map_manager) = &self.map_manager else { return None; };
        
        let mut map_guard = map_manager.write().unwrap();
        let grid_x = world_to_grid_x(pos.x);
        let grid_y = world_to_grid_y(pos.y);
        
        let old_hp = map_guard.get_creature(
            self.current_map_id,
            0,
            grid_x,
            grid_y,
            *guid
        ).map(|c| c.current_hp)?;
        
        let new_hp = old_hp.saturating_sub(damage);
        
        let died = map_guard.with_creature_mut(
            self.current_map_id,
            0,
            grid_x,
            grid_y,
            *guid,
            |creature| {
                creature.current_hp = new_hp;
                if new_hp == 0 {
                    creature.is_alive = false;
                    creature.state = CreatureState::Dead;
                    creature.corpse_despawn_at = Some(std::time::Instant::now() + std::time::Duration::from_secs(30));
                    true // died
                } else {
                    false
                }
            }
        );
        
        if died == Some(true) {
            debug!("Creature {:?} died from damage", guid);
        }
        
        Some(new_hp)
    }

    /// Legacy method: Get mutable reference to creature via closure.
    /// TEMPORARY: This creates a local copy, modifies it, and writes back.
    /// Will be replaced with proper reference-based API after full migration.
    pub(crate) fn with_creature_mut<F, R>(&self, guid: &ObjectGuid, f: F) -> Option<R>
    where
        F: FnOnce(&mut WorldCreature) -> R,
    {
        let Some(pos) = self.player_position else { return None; };
        let Some(map_manager) = &self.map_manager else { return None; };
        
        let grid_x = world_to_grid_x(pos.x);
        let grid_y = world_to_grid_y(pos.y);
        
        // Read creature
        let creature = {
            let map_guard = map_manager.read().unwrap();
            map_guard.get_creature(
                self.current_map_id,
                0,
                grid_x,
                grid_y,
                *guid
            ).cloned()
        }?;
        
        // Modify
        let mut mutable = creature;
        let result = f(&mut mutable);
        
        // Write back
        let mut map_guard = map_manager.write().unwrap();
        map_guard.add_creature(
            self.current_map_id,
            0,
            grid_x,
            grid_y,
            mutable
        );
        
        Some(result)
    }

    /// Legacy method: Get creature reference via closure.
    pub(crate) fn with_creature<F, R>(&self, guid: &ObjectGuid, f: F) -> Option<R>
    where
        F: FnOnce(&WorldCreature) -> R,
    {
        let Some(pos) = self.player_position else { return None; };
        let Some(map_manager) = &self.map_manager else { return None; };
        
        let map_guard = map_manager.read().unwrap();
        let grid_x = world_to_grid_x(pos.x);
        let grid_y = world_to_grid_y(pos.y);
        
        map_guard.get_creature(
            self.current_map_id,
            0,
            grid_x,
            grid_y,
            *guid
        ).map(f)
    }
    
    /// Legacy method: Get creature directly (returns a clone).
    pub(crate) fn get_creature(&self, guid: &ObjectGuid) -> Option<WorldCreature> {
        self.with_creature(guid, |c| c.clone())
    }
